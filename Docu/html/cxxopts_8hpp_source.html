656 inline OptionNames split_option_names(const std::string &text)
657 {
658  OptionNames split_names;
659 
660  std::string::size_type token_start_pos = 0;
661  auto length = text.length();
662 
663  if (length == 0)
664  {
665  throw_or_mimic<exceptions::invalid_option_format>(text);
666  }
667 
668  while (token_start_pos < length) {
669  const auto &npos = std::string::npos;
670  auto next_non_space_pos = text.find_first_not_of(' ', token_start_pos);
671  if (next_non_space_pos == npos) {
672  throw_or_mimic<exceptions::invalid_option_format>(text);
673  }
674  token_start_pos = next_non_space_pos;
675  auto next_delimiter_pos = text.find(',', token_start_pos);
676  if (next_delimiter_pos == token_start_pos) {
677  throw_or_mimic<exceptions::invalid_option_format>(text);
678  }
679  if (next_delimiter_pos == npos) {
680  next_delimiter_pos = length;
681  }
682  auto token_length = next_delimiter_pos - token_start_pos;
683  // validate the token itself matches the regex /([:alnum:][-_[:alnum:]]*/
684  {
685  const char* option_name_valid_chars =
686  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
687  "abcdefghijklmnopqrstuvwxyz"
688  "0123456789"
689  "_-.?";
690 
691  if (!std::isalnum(text[token_start_pos], std::locale::classic()) ||
692  text.find_first_not_of(option_name_valid_chars, token_start_pos) < next_delimiter_pos) {
693  throw_or_mimic<exceptions::invalid_option_format>(text);
694  }
695  }
696  split_names.emplace_back(text.substr(token_start_pos, token_length));
697  token_start_pos = next_delimiter_pos + 1;
698  }
699  return split_names;
700 }

703 {
704  ArguDesc argu_desc;
705  const char *pdata = arg;
706  matched = false;
707  if (strncmp(pdata, "--", 2) == 0)
708  {
709  pdata += 2;
710  if (isalnum(*pdata, std::locale::classic()))
711  {
712  argu_desc.arg_name.push_back(*pdata);
713  pdata += 1;
714  while (isalnum(*pdata, std::locale::classic()) || *pdata == '-' || *pdata == '_')
715  {
716  argu_desc.arg_name.push_back(*pdata);
717  pdata += 1;
718  }
719  if (argu_desc.arg_name.length() > 1)
720  {
721  if (*pdata == '=')
722  {
723  argu_desc.set_value = true;
724  pdata += 1;
725  if (*pdata != '\0')
726  {
727  argu_desc.value = std::string(pdata);
728  }
729  matched = true;
730  }
731  else if (*pdata == '\0')
732  {
733  matched = true;
734  }
735  }
736  }
737  }
738  else if (strncmp(pdata, "-", 1) == 0)
739  {
740  pdata += 1;
741  argu_desc.grouping = true;
742  while (isalnum(*pdata, std::locale::classic()))
743  {

745  pdata += 1;
746  }
747  matched = !argu_desc.arg_name.empty() && *pdata == '\0';
748  }

921 template <typename T>
922 void
923 integer_parser(const std::string& text, T& value)
924 {
925  parser_tool::IntegerDesc int_desc = parser_tool::SplitInteger(text);
926 
927  using US = typename std::make_unsigned<T>::type;
928  constexpr bool is_signed = std::numeric_limits<T>::is_signed;
929 
930  const bool negative = int_desc.negative.length() > 0;
931  const uint8_t base = int_desc.base.length() > 0 ? 16 : 10;
932  const std::string & value_match = int_desc.value;
933 
934  US result = 0;
935 
936  for (char ch : value_match)
937  {
938  US digit = 0;
939 
940  if (ch >= '0' && ch <= '9')
941  {
942  digit = static_cast<US>(ch - '0');
943  }
944  else if (base == 16 && ch >= 'a' && ch <= 'f')
945  {
946  digit = static_cast<US>(ch - 'a' + 10);
947  }
948  else if (base == 16 && ch >= 'A' && ch <= 'F')
949  {
950  digit = static_cast<US>(ch - 'A' + 10);
951  }
952  else
953  {
954  throw_or_mimic<exceptions::incorrect_argument_type>(text);
955  }
956 
957  const US next = static_cast<US>(result * base + digit);
958  if (result > next)
959  {
960  throw_or_mimic<exceptions::incorrect_argument_type>(text);
961  }
962 
963  result = next;
964  }
965 
966  detail::check_signed_range<T>(negative, result, text);
967 
968  if (negative)
969  {
970  checked_negate<T>(value, result, text, std::integral_constant<bool, is_signed>());
971  }
972  else
973  {
974  value = static_cast<T>(result);
975  }
976 }
